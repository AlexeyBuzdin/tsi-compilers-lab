procedure TEditor.NewLine;
const
  CrLf: array[1..2] of Char = #13#10;
var  I, P: Word;
begin
  P := LineStart(CurPtr);  I := P;
  while (I < CurPtr) and ((Buffer^[I] = ' ') or (Buffer^[I] = #9)) do Inc(I);
  InsertText(@CrLf, 2, False);
  if AutoIndent then InsertText(@Buffer^[P], I - P, False);
end;
